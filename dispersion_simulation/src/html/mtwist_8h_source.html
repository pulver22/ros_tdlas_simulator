00154 /*
00155  * Internal state for an MT RNG.  The user can keep multiple mt_state
00156  * structures around as a way of generating multiple streams of random
00157  * numbers.
00158  *
00159  * In Matsumoto and Nishimura's original paper, the state vector was
00160  * processed in a forward direction.  I have reversed the state vector
00161  * in this implementation.  The reason for the reversal is that it
00162  * allows the critical path to use a test against zero instead of a
00163  * test against 624 to detect the need to refresh the state.  on most
00164  * machines, testing against zero is slightly faster.  It also means
00165  * that a state that has been set to all zeros will be correctly
00166  * detected as needing initialization; this means that setting a state
00167  * vector to zero (either with memset or by statically allocating it)
00168  * will cause the RNG to operate properly.
00169  */

00306  * Tempering parameters.  These are perhaps the most magic of all the magic
00307  * values in the algorithm.  The values are again experimentally determined.
00308  * The values generated by the recurrence relation (constants above) are not
00309  * equidistributed in 623-space.  For some reason, the tempering process
00310  * produces that effect.  Don't ask me why.  Read the paper if you can
00311  * understand the math.  Or just trust these magic numbers.
00312  */
00313 #define MT_TEMPERING_MASK_B 0x9d2c5680
00314 #define MT_TEMPERING_MASK_C 0xefc60000
00315 #define MT_TEMPERING_SHIFT_U(y) \
00316                         (y >> 11)
00317 #define MT_TEMPERING_SHIFT_S(y) \
00318                         (y << 7)
00319 #define MT_TEMPERING_SHIFT_T(y) \
00320                         (y << 15)
00321 #define MT_TEMPERING_SHIFT_L(y) \
00322                         (y >> 18)

00324 /*
00325  * Macros to do the tempering.  MT_PRE_TEMPER does all but the last step;
00326  * it's useful for situations where the final step can be incorporated
00327  * into a return statement.  MT_FINAL_TEMPER does that final step (not as
00328  * an assignment).  MT_TEMPER does the entire process.  Note that
00329  * MT_PRE_TEMPER and MT_TEMPER both modify their arguments.
00330  */

00332     do                                                                  \
00333         {                                                               \

00380 /*
00381  * Generate a random number in the range 0 to 2^32-1, inclusive, working
00382  * from a given state vector.
00383  *
00384  * The generator is optimized for speed.  The primary optimization is that
00385  * the pseudorandom numbers are generated in batches of MT_STATE_SIZE.  This
00386  * saves the cost of a modulus operation in the critical path.
00387  */
00388 MT_EXTERN MT_INLINE unsigned long mts_lrand(
00389     register mt_state*  state)          /* State for the PRNG */
00390 {
00391         register unsigned long
00392         random_value;   /* Pseudorandom value generated */
00393 
00394         if (state->stateptr <= 0)
00395                 mts_refresh(state);
00396 
00397         random_value = state->statevec[--state->stateptr];
00398         MT_PRE_TEMPER(random_value);
00399         return MT_FINAL_TEMPER(random_value);
00400 }

00403 /*
00404  * Generate a random number in the range 0 to 2^64-1, inclusive, working
00405  * from a given state vector.
00406  *
00407  * According to Matsumoto and Nishimura, such a number can be generated by
00408  * simply concatenating two 32-bit pseudorandom numbers.  Who am I to argue?
00409  *
00410  * Note that there is a slight inefficiency here: if the 624-entry state is
00411  * recycled on the second call to mts_lrand, there will be an unnecessary
00412  * check to see if the state has been initialized.  The cost of that check
00413  * seems small (since it happens only once every 624 random numbers, and
00414  * never if only 64-bit numbers are being generated), so I didn't bother to
00415  * optimize it out.  Doing so would be messy, since it would require two
00416  * nearly-identical internal implementations of mts_lrand.
00417  */
00418 MT_EXTERN MT_INLINE unsigned long long mts_llrand(
00419     register mt_state*  state)          /* State for the PRNG */
00420 {
00421         register unsigned long
00422         random_value_1; /* 1st pseudorandom value generated */
00423         register unsigned long
00424         random_value_2; /* 2nd pseudorandom value generated */
00425 
00426         /*
00427          * For maximum speed, we'll handle the two overflow cases
00428          * together.  That will save us one test in the common case, at
00429          * the expense of an extra one in the overflow case.
00430          */
00431         if (--state->stateptr <= 0)
00432         {
00433                 if (state->stateptr < 0)
00434                 {
00435                         mts_refresh(state);
00436                         random_value_1 = state->statevec[--state->stateptr];
00437                 }
00438                 else
00439                 {
00440                         random_value_1 = state->statevec[state->stateptr];
00441                         mts_refresh(state);
00442                 }
00443         }
00444         else
00445                 random_value_1 = state->statevec[--state->stateptr];
00446 
00447         MT_TEMPER(random_value_1);
00448 
00449         random_value_2 = state->statevec[--state->stateptr];
00450         MT_PRE_TEMPER(random_value_2);
00451 
00452         return ((unsigned long long) random_value_1 << 32)
00453                | (unsigned long long) MT_FINAL_TEMPER(random_value_2);
00454 }

00615 /*
00616  * Generate a double-precision random number between 0 (inclusive) and 1.0
00617  * (exclusive).  This function generates 64 bits of precision.  Use
00618  * mts_drand for more speed but less precision.
00619  */
00620 MT_EXTERN MT_INLINE double mt_ldrand(void)
00621 {
00622 #if MT_MACHINE_BITS == 64
00623         unsigned long long      final_value;    /* Final (integer) value */
00624 #endif /* MT_MACHINE_BITS */
00625         register unsigned long
00626         random_value_1; /* 1st pseudorandom value generated */
00627         register unsigned long
00628         random_value_2; /* 2nd pseudorandom value generated */
00629 
00630         /*
00631          * For maximum speed, we'll handle the two overflow cases
00632          * together.  That will save us one test in the common case, at
00633          * the expense of an extra one in the overflow case.
00634          */
00635         if (--mt_default_state.stateptr <= 0)
00636         {
00637                 if (mt_default_state.stateptr < 0)
00638                 {
00639                         mts_refresh(&mt_default_state);
00640                         random_value_1 =
00641                             mt_default_state.statevec[--mt_default_state.stateptr];
00642                 }
00643                 else
00644                 {
00645                         random_value_1 =
00646                             mt_default_state.statevec[mt_default_state.stateptr];
00647                         mts_refresh(&mt_default_state);
00648                 }
00649         }
00650         else
00651                 random_value_1 =
00652                     mt_default_state.statevec[--mt_default_state.stateptr];
00653 
00654         MT_TEMPER(random_value_1);
00655 
00656         random_value_2 = mt_default_state.statevec[--mt_default_state.stateptr];
00657         MT_TEMPER(random_value_2);
00658 
00659 #if MT_MACHINE_BITS == 64
00660         final_value = ((unsigned long long) random_value_1 << 32)
00661                       | (unsigned long long) random_value_2;
00662         return final_value * mt_64_to_double;
00663 #else /* MT_MACHINE_BITS */
00664         return random_value_1 * mt_32_to_double + random_value_2 * mt_64_to_double;
00665 #endif /* MT_MACHINE_BITS */
